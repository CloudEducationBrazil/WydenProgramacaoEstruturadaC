#include <stdio.h>

int main() {
    int dia, mes, diasPassados;

    printf("--- Calculador de Dias Decorridos ---\n");

    
    printf("Digite o dia (1-30): ");
    scanf("%d", &dia);

    
    printf("Digite o mes (1-12): ");
    scanf("%d", &mes);

    /* Lógica: Se estamos no mês 3, passaram-se 2 meses completos (janeiro e fevereiro).
       Portanto, multiplicamos (mes - 1) por 30 e somamos os dias do mês atual.
    */
    diasPassados = ((mes - 1) * 30) + dia;

    
    printf("\nDesde o inicio do ano, se passaram %d dias.\n", diasPassados);

    return 0;
}

    float conta1 = 200.00;

}
